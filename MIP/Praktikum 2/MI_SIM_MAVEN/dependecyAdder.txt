Before packaging build these statements should be excecuted to add all local jars to the local repository.



mvn install:install-file -Dfile=slick.jar \
                         -DgroupId=org.newdawn\
                         -DartifactId=slick \
                         -Dversion=1.0 \
                         -Dpackaging=jar

mvn install:install-file -Dfile=lwjgl.jar \
                         -DgroupId=org.lwjgl\
                         -DartifactId=lwjgl \
                         -Dversion=1.0 \
                         -Dpackaging=jar
                         
mvn install:install-file -Dfile=lwjgl_util.jar \
                         -DgroupId=org.lwjgl\
                         -DartifactId=lwjgl_util \
                         -Dversion=1.0 \
                         -Dpackaging=jar
                         
mvn install:install-file -Dfile=natives-linux.jar \
                         -DgroupId=org.lwjgl\
                         -DartifactId=nativeslinux \
                         -Dversion=1.0 \
                         -Dpackaging=jar
